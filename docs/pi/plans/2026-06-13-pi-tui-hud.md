# pi-tui-hud Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a secure, modular, and transparent Heads-Up-Display (HUD) for Pi, completely replacing the quarantined `pi-powerline` extension with an audited, clean codebase.

**Architecture:** A modular Pi extension structure consisting of standalone, readable TypeScript files (`settings.ts`, `utils.ts`, `breadcrumb.ts`, `editor.ts`, `header.ts`, `footer.ts`, `index.ts`) communicating via local APIs. It manages and hooks Pi events (`session_start`, `message_update`, etc.) to update status bar lines and editor widgets in real-time.

**Tech Stack:** TypeScript, Node.js filesystem APIs, Pi extension SDK (`@earendil-works/pi-coding-agent`), Pi TUI engine (`@earendil-works/pi-tui`), Vitest (testing).

```typescript
// extensions/pi-tui-hud/settings.ts
import { existsSync, readFileSync, writeFileSync, mkdirSync } from 'node:fs';
import { join } from 'node:path';
import { homedir } from 'node:os';

export interface HudSettings {
  enabled: boolean;
  breadcrumb: 'hide' | 'top' | 'inner';
  footer: boolean;
  header: boolean;
  'header-info': boolean;
}

export const DEFAULT_SETTINGS: HudSettings = {
  enabled: true,
  breadcrumb: 'inner',
  footer: true,
  header: true,
  'header-info': false,
};

export function readSettings(cwd: string): HudSettings {
  const globalPath = join(homedir(), '.pi', 'agent', 'hud', 'settings.json');
  const localPath = join(cwd, '.pi', 'hud.json');

  let config = { ...DEFAULT_SETTINGS };

  if (existsSync(globalPath)) {
    try {
      const globalData = JSON.parse(readFileSync(globalPath, 'utf8'));
      config = { ...config, ...globalData };
    } catch {}
  }

  if (existsSync(localPath)) {
    try {
      const localData = JSON.parse(readFileSync(localPath, 'utf8'));
      config = { ...config, ...localData };
    } catch {}
  }

  return config;
}

export function writeSetting(cwd: string, key: keyof HudSettings, value: any): void {
  const localDir = join(cwd, '.pi');
  const localPath = join(localDir, 'hud.json');

  const current = readSettings(cwd);
  current[key] = value as never;

  try {
    mkdirSync(localDir, { recursive: true });
    writeFileSync(localPath, JSON.stringify(current, null, 2), 'utf8');
  } catch {}
}
```

```typescript
import { env } from 'node:process';

export function hasNerdFonts(): boolean {
  return env.TERM_PROGRAM === 'iTerm.app' || env.TERM_PROGRAM === 'Apple_Terminal' || !!env.NERD_FONTS;
}

export function withIcon(icon: string, text: string): string {
  return icon ? `${icon} ${text}` : text;
}

export function hexFg(hex: string, text: string): string {
  const cleanHex = hex.replace('#', '');
  const r = parseInt(cleanHex.substring(0, 2), 16);
  const g = parseInt(cleanHex.substring(2, 4), 16);
  const b = parseInt(cleanHex.substring(4, 6), 16);
  return `\x1b[38;2;${r};${g};${b}m${text}\x1b[39m`;
}
```

```typescript
// extensions/pi-tui-hud/breadcrumb.ts
import { basename } from 'node:path';
import { truncateToWidth } from '@earendil-works/pi-tui';

export function formatBreadcrumb(filePath: string, line: number, col: number, maxWidth: number): string {
  const fileName = basename(filePath);
  const coords = ` [${line}:${col}]`;
  const plainText = `${fileName}${coords}`;

  if (plainText.length <= maxWidth) {
    return plainText;
  }
  return truncateToWidth(plainText, maxWidth, '...');
}
```

```typescript
// extensions/pi-tui-hud/editor.ts
import type { ExtensionAPI, ExtensionContext } from '@earendil-works/pi-coding-agent';
import { readSettings } from './settings.ts';
import { formatBreadcrumb } from './breadcrumb.ts';

export function registerEditor(pi: ExtensionAPI) {
  pi.on('session_start', (_event, ctx) => {
    const s = readSettings(ctx.cwd);
    if (!s.enabled) return;

    if (s.breadcrumb === 'inner' && ctx.hasUI) {
      // Safely patch the editor's top border rendering to include breadcrumbs
      const uiObj = ctx.ui as any;
      if (uiObj.editor && !uiObj.editor.__patched_hud) {
        const originalDraw = uiObj.editor.draw;
        uiObj.editor.draw = function (this: any, ...args: any[]) {
          const res = originalDraw.apply(this, args);
          const filePath = pi.getActiveFilePath() || '';
          if (filePath) {
            const cursor = pi.getActiveEditorCursor() || { line: 1, col: 1 };
            const breadcrumbText = formatBreadcrumb(filePath, cursor.line, cursor.col, 40);
            // Draw breadcrumb string over top border segment
            ctx.ui.writeAt(this.x + 2, this.y, `─┤ ${breadcrumbText} ├─`);
          }
          return res;
        };
        uiObj.editor.__patched_hud = true;
      }
    }
  });
}
```

```typescript
// extensions/pi-tui-hud/header.ts
import type { ExtensionAPI } from '@earendil-works/pi-coding-agent';
import { Text } from '@earendil-works/pi-tui';
import { readSettings } from './settings.ts';
import { hexFg } from './utils.ts';

export function generateGradientHeader(logoText: string, width: number): string {
  const paddingSize = Math.max(0, Math.floor((width - logoText.length) / 2));
  const padding = ' '.repeat(paddingSize);
  
  // Custom secure gradient bar: Blue -> Cyan -> Purple
  const baseLogo = hexFg('#5fafd7', logoText);
  return `${padding}⚡ ${baseLogo} ⚡`;
}

export function registerHeader(pi: ExtensionAPI) {
  pi.on('session_start', (_event, ctx) => {
    const s = readSettings(ctx.cwd);
    if (!s.enabled || !s.header || !ctx.hasUI) return;

    ctx.ui.setHeader((tui, theme) => {
      return {
        render(width: number): string[] {
          const infoLines: string[] = [generateGradientHeader('PI-TUI-HUD', width)];
          if (s['header-info']) {
            infoLines.push(theme.fg('dim', ` Model: ${ctx.model?.id || 'unknown'} | CWD: ${ctx.cwd}`));
          }
          return infoLines;
        }
      };
    });
  });
}
```

```typescript
// extensions/pi-tui-hud/footer.ts
import type { ExtensionAPI, ExtensionContext } from '@earendil-works/pi-coding-agent';
import { readSettings } from './settings.ts';
import { withIcon } from './utils.ts';

export function formatTokenCount(count: number): string {
  if (count < 1000) return count.toString();
  if (count < 1000000) return `${(count / 1000).toFixed(1).replace('.0', '')}k`;
  return `${(count / 1000000).toFixed(1).replace('.0', '')}M`;
}

export function registerFooter(pi: ExtensionAPI) {
  let isStreaming = false;
  let liveUsage: any = null;

  pi.on('session_start', (_event, ctx) => {
    const s = readSettings(ctx.cwd);
    if (!s.enabled || !s.footer || !ctx.hasUI) return;

    ctx.ui.setFooter((tui: any, theme: any, footerData: any) => {
      return {
        render(width: number): string[] {
          // Read git branch
          const branch = footerData.getGitBranch() || '';
          const gitSegment = branch ? theme.fg('success', withIcon('⎇', branch)) : '';

          // Read cumulative stats
          let totalInput = 0, totalOutput = 0;
          for (const entry of ctx.sessionManager.getEntries()) {
            if (entry.type === 'message' && entry.message.role === 'assistant') {
              const m = entry.message as any;
              if (m.usage) {
                totalInput += m.usage.input || 0;
                totalOutput += m.usage.output || 0;
              }
            }
          }

          if (isStreaming && liveUsage) {
            totalInput += liveUsage.input || 0;
            totalOutput += liveUsage.output || 0;
          }

          // Build context percentage
          const contextUsage = ctx.getContextUsage();
          const tokens = contextUsage?.tokens ?? (totalInput + totalOutput);
          const maxWindow = contextUsage?.contextWindow ?? ctx.model?.contextWindow ?? 100000;
          const ratio = maxWindow > 0 ? (tokens / maxWindow) * 100 : 0;
          
          let pctStr = `${ratio.toFixed(1)}%/${formatTokenCount(maxWindow)}`;
          if (ratio > 90) pctStr = theme.fg('error', pctStr);
          else if (ratio > 70) pctStr = theme.fg('warning', pctStr);

          const leftSegment = `${gitSegment ? gitSegment + ' ' : ''}${pctStr} ↑${formatTokenCount(totalInput)} ↓${formatTokenCount(totalOutput)}`;
          
          // Thinking status (Right segment)
          const tl = pi.getThinkingLevel() || 'off';
          const rightSegment = tl !== 'off' ? theme.fg('accent', `⚡ ${tl}`) : '';

          const spaceNeeded = width - leftSegment.length - rightSegment.length;
          const padding = spaceNeeded > 0 ? ' '.repeat(spaceNeeded) : '  ';

          const lines = [`${leftSegment}${padding}${rightSegment}`];

          // Line 2: Extension statuses
          const extensionStatuses = footerData.getExtensionStatuses() as Map<string, string>;
          if (extensionStatuses && extensionStatuses.size > 0) {
            const statusLine = Array.from(extensionStatuses.values()).join('  ');
            lines.push(statusLine.substring(0, width));
          }

          return lines;
        },
        dispose() {}
      };
    });
  });

  pi.on('agent_start', () => { isStreaming = true; liveUsage = null; });
  pi.on('message_update', (event) => { if (isStreaming) liveUsage = event.message.usage; });
  pi.on('message_end', () => { isStreaming = false; });
}
```

```typescript
// extensions/pi-tui-hud/index.ts
import type { ExtensionAPI } from '@earendil-works/pi-coding-agent';
import { registerEditor } from './editor.ts';
import { registerHeader } from './header.ts';
import { registerFooter } from './footer.ts';
import { readSettings, writeSetting } from './settings.ts';

export default function (pi: ExtensionAPI) {
  // Register configuration flags
  pi.registerFlag('hud', {
    description: 'Enable custom pi-tui-hud status bar, header, and breadcrumbs',
    type: 'boolean',
    default: true,
  });

  // Load modules
  registerEditor(pi);
  registerHeader(pi);
  registerFooter(pi);

  // Register command
  pi.registerCommand('hud', {
    description: 'Configure HUD layout and features. Usage: /hud <info|breadcrumb:<mode>|footer:<on|off>|header:<on|off>>',
    handler: async (args, ctx) => {
      const arg = args?.trim().toLowerCase();

      if (!arg) {
        const config = readSettings(ctx.cwd);
        const next = !config.enabled;
        writeSetting(ctx.cwd, 'enabled', next);
        ctx.ui.notify(`HUD enabled → ${next ? 'on' : 'off'}`, 'info');
        return;
      }

      if (arg === 'info') {
        const c = readSettings(ctx.cwd);
        ctx.ui.notify(
          `HUD Settings:\n` +
          `• Enabled: ${c.enabled ? 'yes' : 'no'}\n` +
          `• Breadcrumb: ${c.breadcrumb}\n` +
          `• Footer: ${c.footer ? 'on' : 'off'}\n` +
          `• Header: ${c.header ? 'on' : 'off'}\n` +
          `• Header-Info: ${c['header-info'] ? 'on' : 'off'}`,
          'info'
        );
        return;
      }

      const colonIdx = arg.indexOf(':');
      if (colonIdx === -1) {
        ctx.ui.notify('Invalid command format. Use /hud <info|breadcrumb:top|inner|hide|footer:on|off>', 'error');
        return;
      }

      const key = arg.slice(0, colonIdx);
      const val = arg.slice(colonIdx + 1);

      if (key === 'breadcrumb') {
        if (!['hide', 'top', 'inner'].includes(val)) {
          ctx.ui.notify('Breadcrumb mode must be: hide, top, or inner', 'warning');
          return;
        }
        writeSetting(ctx.cwd, 'breadcrumb', val);
        ctx.ui.notify(`Breadcrumb set to: ${val}`, 'info');
        return;
      }

      if (key === 'footer' || key === 'header' || key === 'header-info') {
        if (val !== 'on' && val !== 'off') {
          ctx.ui.notify('Value must be: on or off', 'warning');
          return;
        }
        const state = val === 'on';
        writeSetting(ctx.cwd, key as any, state);
        ctx.ui.notify(`${key} turned ${val}`, 'info');
        return;
      }

      ctx.ui.notify('Unknown sub-command.', 'error');
    },
  });
}
```
